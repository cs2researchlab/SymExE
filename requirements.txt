# ============================================
# Installation Notes

# CRITICAL: Install in this order to avoid conflicts:
#
# 1. Create fresh conda environment:
#    conda create -n symexe python=3.13
#    conda activate symexe
#
# 2. Install angr (this will install most dependencies):
#    pip install angr==9.2.181
#
# 3. Install remaining dependencies:
#    pip install -r requirements.txt
#
# 4. Verify installation:
#    python3 -c "import angr; print(f'angr {angr.__version__}')"
#    Should print: angr 9.2.181
#
# 5. Test SymExE:
#    python3 symexe.py --help

# ============================================
# Troubleshooting
# ============================================

# If you encounter "externally managed environment" error on some systems:
#   pip install --break-system-packages angr==9.2.181
#
# If you encounter z3 conflicts:
#   pip uninstall z3-solver
#   pip install z3-solver>=4.12.0.0
#
# If angr fails to import:
#   pip uninstall angr claripy archinfo pyvex cle ailment
#   pip install angr==9.2.181
#
# For memory issues during installation:
#   pip install --no-cache-dir angr==9.2.181
